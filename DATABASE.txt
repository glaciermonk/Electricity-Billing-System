CREATE DATABASE EBS;

create table CONSUMER_DB(CONSUMER_ID bigint(100) NOT NULL AUTO_INCREMENT, CONSUMER_NAME varchar(500), CONSUMER_PHONE varchar(500), CONSUMER_EMAIL varchar(500), CONSUMER_STATE varchar(500), CONSUMER_CITY varchar(500), CONSUMER_PIN_CODE bigint(100), METER_ID varchar(200), PRIMARY KEY(CONSUMER_ID));

create table BILL_DB(CONSUMER_ID bigint(100), BILL_ID bigint(100) NOT NULL AUTO_INCREMENT, BILL_AMOUNT bigint(100), BILL_DATE DATE, BILL_DUE_DATE DATE, UNITS_CONSUMED bigint(100), BILL_STATUS varchar(100), PAYMENT_METHOD varchar(255), PRIMARY KEY (BILL_ID), FOREIGN KEY(CONSUMER_ID) REFERENCES CONSUMER_DB(CONSUMER_ID) on delete cascade on update cascade);
